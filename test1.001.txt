<script>

// --------------- BACKGROUND COLOR -----------------

function switchStyle(style) {
    const body = document.getElementById("body");

    body.classList.remove("purple", "yellow", "orange", "red");

    if (style === 0) body.classList.add("purple");
    if (style === 1) body.classList.add("yellow");
    if (style === 2) body.classList.add("orange");
    if (style === 3) body.classList.add("red");
}

// --------------- FONT STYLE CHECKBOXES -----------------

function updateTextStyles() {
    const mainText = document.querySelector("main");

    mainText.classList.toggle("bold", document.getElementById("bold").checked);
    mainText.classList.toggle("italic", document.getElementById("italics").checked);
    mainText.classList.toggle("underline", document.getElementById("underline").checked);
}

// --------------- FONT SIZE DROPDOWN -----------------

function selectFontSize() {
    const mainText = document.querySelector("main");
    const size = document.getElementById("fonts").value;

    mainText.classList.remove("small", "medium", "large");
    mainText.classList.add(size);
}

</script>
